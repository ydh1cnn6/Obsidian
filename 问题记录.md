private String justify(String s) {  
    if (width == -1)  
        return s;  
    StringBuilder sb = new StringBuilder();  
    //f:待插入的格式，不写时f="",0时f=0
    boolean pad = f.contains(Flags.LEFT_JUSTIFY);  
    int sp = width - s.length();  
    if (!pad)  
        for (int i = 0; i < sp; i++) sb.append(' ');  
    sb.append(s);  
    if (pad)  
        for (int i = 0; i < sp; i++) sb.append(' ');  
    return sb.toString();  
}
